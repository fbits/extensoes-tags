1-Ter acessar ao gerenciador de scripts do Admin Fstore
2-Adicionar nova TAG, com o  nome "Fbits.Tag.Lomadee" 
3-Determinar os prazos de exibição, caso fique sempre deixar os campos em branco. 
4-Deixar ativo e prioridade 1
5-Posição deixar como "Header - Última linha"
6-Página colocar todas. E no Identificador colocar "*"
7-No campo conteúdo do script adicionar o script abaixo.
8-Substituir ADVERTISERID com o ID do Anunciante na Lomadee.
9-Substituir TEMPO com o a validade em dias para expirar o cookie

<Script>
if((window.location.search.indexOf("utm=Lomadee") > -1) ||
	(window.location.search.indexOf("utm_source=Lomadee") > -1) ||
	(window.location.search.indexOf("source=Lomadee") > -1))
{   
	Fbits.Cookie.Set('parceiroAtivo', 'Lomadee', TEMPO);

	}
	
	var parceiroAtivo =  Fbits.Cookie.Get('ParceiroAtivo');	
	if (parceiroAtivo == 'Lomadee'){


	//Lomadee otimização
	try{var oppuzJSProtocol = (("https:" == document.location.protocol) ? "https://" : "http://"); 
		var head= document.getElementsByTagName('head')[0]; 
		var script= document.createElement('script'); 
		script.type= 'text/javascript'; 
		script.src= oppuzJSProtocol + "www.oppuz.com/script/lmd/ADVERTISERID.js"; 
		script.async = true; 
		head.appendChild(script); } catch (e) {}
	}
}
</Script>
