3-Determinar os prazos de exibição, caso fique sempre deixar os campos em branco. 
4-Deixar ativo e prioridade 2
5-Posição deixar como "Header - Última linha"
6-Página colocar "Home 2". E no Identificador colocar "*"
7-No campo conteúdo do script adicionar o script abaixo.

    <script type="text/javascript">
        var parceiroAtivo = Fbits.Cookie.Get("ParceiroAtivo").toLowerCase();
        if (parceiroAtivo == 'criteo') {
            (function () {
                var xscr = document.createElement('script');
                xscr.async = true;
                xscr.src = '//static.criteo.net/js/ld/ld.js';
                var x = document.getElementById('criteo');
                x.parentNode.insertBefore(xscr, x);
            }());
            window.criteo_q = window.criteo_q || [];
            window.criteo_q.push(
                { event: "setAccount", account: 4160 },
                { event: "setEmail", email: [""] },
                { event: "setSiteType", type: "d" },
                { event: "viewHome" }
            );
        }
    </script>
