1-Ter acessar ao gerenciador de scripts do Admin Fstore
2-Adicionar nova TAG, com o  nome "Fbits.Tag.AllIn.viewCarrinho" 
3-Determinar os prazos de exibição, caso fique sempre deixar os campos em branco. 
4-Deixar ativo e prioridade 10
5-Posição deixar como "Body - Última linha"
6-Página colocar "Todas". 
7-E no Identificador colocar "*Carrinho*"
8-No campo conteúdo do script adicionar o script abaixo.


<script type="text/javascript">
    console.log("Listando: Allin.removeProdutoCarrinho");
    $("#btnRemover-@itemId").bind('click', function () {//faz um bind no click do botão remover chamando o metodo addProdutoCarrinho passando 0 na quantidade
        try {
            console.log("Click: Allin.removeProdutoCarrinho");
            bs.addProdutoCarrinho({
                "id_produto": "" + Fbits.Produto.ProdutoId + "",/*Id Produto, OBRIGATÓRIO*/
                "nome_produto": "" + Fbits.Usuario.Nome + "",/*Nome Produto, OBRIGATÓRIO*/
                "valor_unitario": "" + Fbits.Produto.PrecoPor + "",/*Valor do produto, OBRIGATÓRIO*/
                "quantidade": "0",/*Quantidade do produto, NÃO OBRIGATÓRIO */
                "categoria": "" + Fbits.Produto.Categorias[0].CategoriaId + "",/*Categoria, OBRIGATÓRIO*/ //(PROVAVELMENTE EU TENHO QUE FAZER UM LYNC pra pegar a categoria)
                "categoria_sub": "" + Fbits.Produto.Categorias[0].CategoriaId + "",/*SubCategoria do Produto, OBRIGATÓRIO *///(e outro para pegar a subcategoria)
                "marca": "" + Fbits.Produto.Marca + ""/*Marca (ou fabricante), OBRIGATÓRIO*/
            });
        } catch (e) {
            //erros
        }
    });
</script>
